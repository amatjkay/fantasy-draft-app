    });

    } else {
      this.sortColumn = column;
      this.sortDirection = 'asc';
    }
    
    this.render();
  }

      this.filteredData = this.data.filter(p => p.position === position);
    }
    this.render();
  }

// Salary Cap tracker
class SalaryCapTracker {
  constructor(cap = 95500000) {
    this.cap = cap;
    this.used = 0;
    this.update();
  }
  
  add(salary) {
    if (this.used + salary > this.cap) {
      return false;
    }
    this.used += salary;
    this.update();
    return true;
  }
  
  remove(salary) {
    this.used -= salary;
    this.update();
  }
  
  update() {
    const percentage = (this.used / this.cap * 100).toFixed(1);
    document.querySelector('.percentage').textContent = `${percentage}%`;
    document.querySelector('.salary-cap-fill').style.width = `${percentage}%`;
    document.querySelector('.used').textContent = this.formatMoney(this.used);
    document.querySelector('.remaining strong').textContent = this.formatMoney(this.cap - this.used);
  }
  
  formatMoney(amount) {
    return `$${amount.toLocaleString('en-US')}`;
  }
}

// Draft Timer
class DraftTimer {
  constructor(duration = 60) {
    this.duration = duration;
    this.remaining = duration;
    this.interval = null;
    this.callbacks = [];
  }
  
  start() {
    this.interval = setInterval(() => {
      this.remaining--;
      this.update();
      
      if (this.remaining <= 0) {
        this.stop();
        this.onTimeout();
      }
    }, 1000);
  }
  
  stop() {
    clearInterval(this.interval);
  }
  
  reset() {
    this.stop();
    this.remaining = this.duration;
    this.update();
  }
  
  update() {
    const minutes = Math.floor(this.remaining / 60);
    const seconds = this.remaining % 60;
    const display = `${minutes}:${seconds.toString().padStart(2, '0')}`;
    
    document.querySelector('.timer-value').textContent = display;
    
    if (this.remaining <= 10) {
      document.querySelector('.timer-widget').dataset.warning = 'true';
    }
  }
  
  onTimeout() {
    // Auto-pick logic
    this.callbacks.forEach(cb => cb());
  }
  
  addCallback(callback) {
    this.callbacks.push(callback);
  }
}
